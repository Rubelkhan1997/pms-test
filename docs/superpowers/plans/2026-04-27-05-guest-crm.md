# Guest CRM Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Golden Guest Profile system — a single source of truth for guest identity, preferences, compliance data, VIP status, and stay history, with deduplication, PII encryption, and GDPR anonymization support.

**Architecture:** `id_number` encrypted at rest via Laravel `Encrypted` cast. Deduplication engine checks email + phone + id_number before creating. Stay history updated asynchronously on checkout via Observer. Preferences auto-injected into new reservations.

**Tech Stack:** Laravel 13, AES-256 via `Encrypted` cast, Pest PHP.

**Depends on:** Phase 3 (Reservation Engine — guest_id FK on reservations).

declare(strict_types=1);

namespace App\Modules\Guest\Models;

use App\Enums\GuestStatus;
use Illuminate\Database\Eloquent\Model;
use Illuminate\Database\Eloquent\Relations\HasMany;
use Illuminate\Database\Eloquent\SoftDeletes;

class GuestProfile extends Model
{
    use SoftDeletes;

    protected $table = 'guest_profiles';

    protected $fillable = [
        'first_name', 'last_name', 'email', 'mobile', 'gender', 'dob',
        'nationality', 'address', 'city', 'country', 'postal_code',
        'id_type', 'id_number_encrypted', 'id_issue_country', 'id_expiry_date', 'id_document_path',
        'bedding_preference', 'smoking_preference', 'floor_preference',
        'allergies', 'special_requests',
        'status', 'total_stays', 'total_revenue', 'last_stay_date', 'blacklist_reason',
    ];

    protected $casts = [
        'dob'               => 'date',
        'id_expiry_date'    => 'date',
        'last_stay_date'    => 'date',
        'smoking_preference'=> 'boolean',
        'status'            => GuestStatus::class,
        'total_revenue'     => 'decimal:2',
        // AES-256 encryption via Laravel's built-in Encrypted cast
        'id_number_encrypted' => 'encrypted',
    ];

    public function fullName(): string
    {
        return "{$this->first_name} {$this->last_name}";
    }

    public function notes(): HasMany
    {
        return $this->hasMany(GuestNote::class, 'guest_id');
    }

    public function scopeSearch(\Illuminate\Database\Eloquent\Builder $query, string $term): \Illuminate\Database\Eloquent\Builder
    {
        return $query->where(function ($q) use ($term): void {
            $q->where('first_name', 'like', "%{$term}%")
              ->orWhere('last_name', 'like', "%{$term}%")
              ->orWhere('email', 'like', "%{$term}%")
              ->orWhere('mobile', 'like', "%{$term}%");
        });
    }
}

<?php

declare(strict_types=1);

namespace App\Modules\Guest\Actions;

use App\Modules\Guest\Models\GuestProfile;

class FindOrCreateGuestAction
{
    /**
     * @return array{guest: GuestProfile, is_duplicate: bool}
     */
    public function findOrCreate(array $data): array
    {
        // Dedup: check email, mobile, or id_number (cannot dedup encrypted field in SQL, load & compare)
        $existing = null;

        if (! empty($data['email'])) {
            $existing = GuestProfile::where('email', $data['email'])->first();
        }

        if (! $existing && ! empty($data['mobile'])) {
            $existing = GuestProfile::where('mobile', $data['mobile'])->first();
        }

        if ($existing) {
            return ['guest' => $existing, 'is_duplicate' => true];
        }

        $guest = GuestProfile::create($data);

        return ['guest' => $guest, 'is_duplicate' => false];
    }
}

<?php

declare(strict_types=1);

namespace App\Modules\Guest\Services;

use App\Enums\GuestStatus;
use App\Modules\Guest\Actions\AnonymizeGuestAction;
use App\Modules\Guest\Actions\FindOrCreateGuestAction;
use App\Modules\Guest\Models\GuestProfile;
use Illuminate\Pagination\LengthAwarePaginator;

readonly class GuestService
{
    public function __construct(
        private FindOrCreateGuestAction $findOrCreate,
        private AnonymizeGuestAction    $anonymize,
    ) {}

    public function paginate(array $filters = []): LengthAwarePaginator
    {
        return GuestProfile::query()
            ->when($filters['status'] ?? null, fn ($q, $s) => $q->where('status', $s))
            ->when($filters['search'] ?? null, fn ($q, $s) => $q->search($s))
            ->latest()
            ->paginate(20);
    }

    public function create(array $data): array
    {
        return $this->findOrCreate->findOrCreate($data);
    }

    public function update(GuestProfile $guest, array $data): GuestProfile
    {
        $guest->update($data);
        return $guest->refresh();
    }

    public function setVIP(GuestProfile $guest): void
    {
        $guest->update(['status' => GuestStatus::VIP]);
    }

    public function blacklist(GuestProfile $guest, string $reason): void
    {
        $guest->update(['status' => GuestStatus::Blacklisted, 'blacklist_reason' => $reason]);
    }

    public function anonymize(GuestProfile $guest): void
    {
        $this->anonymize->execute($guest);
    }
}

<?php

declare(strict_types=1);

namespace App\Modules\Guest\Controllers\Api\V1;

use App\Http\Controllers\Controller;
use App\Modules\Guest\Models\GuestProfile;
use App\Modules\Guest\Requests\StoreGuestRequest;
use App\Modules\Guest\Resources\GuestResource;
use App\Modules\Guest\Services\GuestService;
use Illuminate\Http\JsonResponse;
use Illuminate\Http\Request;

class GuestController extends Controller
{
    public function __construct(private readonly GuestService $service) {}

    public function index(Request $request): JsonResponse
    {
        $guests = $this->service->paginate($request->only('status', 'search'));
        return response()->json(['status' => 1, 'data' => GuestResource::collection($guests), 'message' => '']);
    }

    public function store(StoreGuestRequest $request): JsonResponse
    {
        $result = $this->service->create($request->validated());
        return response()->json([
            'status'       => 1,
            'data'         => new GuestResource($result['guest']),
            'message'      => $result['is_duplicate'] ? 'Existing guest found.' : 'Guest created.',
            'is_duplicate' => $result['is_duplicate'],
        ], $result['is_duplicate'] ? 200 : 201);
    }

    public function show(GuestProfile $guestProfile): JsonResponse
    {
        $guestProfile->load('notes');
        return response()->json(['status' => 1, 'data' => new GuestResource($guestProfile), 'message' => '']);
    }

    public function update(StoreGuestRequest $request, GuestProfile $guestProfile): JsonResponse
    {
        $guest = $this->service->update($guestProfile, $request->validated());
        return response()->json(['status' => 1, 'data' => new GuestResource($guest), 'message' => 'Updated.']);
    }

    public function setVip(GuestProfile $guestProfile): JsonResponse
    {
        $this->service->setVIP($guestProfile);
        return response()->json(['status' => 1, 'data' => new GuestResource($guestProfile->refresh()), 'message' => 'VIP status set.']);
    }

    public function blacklist(Request $request, GuestProfile $guestProfile): JsonResponse
    {

Create `resources/js/Stores/Guest/guestStore.ts`:

```typescript
import { defineStore } from 'pinia'
import { ref } from 'vue'
import apiClient from '@/Services/apiClient'
import { mapGuest } from '@/Utils/Mappers/guest'
import type { Guest } from '@/Types/Guest/guest'

export const useGuestStore = defineStore('guest', () => {
  const guests        = ref<Guest[]>([])
  const current       = ref<Guest | null>(null)
  const loading       = ref(false)
  const loadingList   = ref(false)
  const loadingDetail = ref(false)

  async function fetchAll(params: Record<string, unknown> = {}, forceRefresh = false): Promise<void> {
    if (guests.value.length && !forceRefresh) return
    loadingList.value = true
    try {
      const res = await apiClient.get('/api/v1/guests', { params })
      guests.value = res.data.data.data.map(mapGuest)
    } finally {
      loadingList.value = false
    }
  }

  async function fetchOne(id: number): Promise<void> {
    loadingDetail.value = true
    try {
      const res = await apiClient.get(`/api/v1/guests/${id}`)
      current.value = mapGuest(res.data.data)
    } finally {
      loadingDetail.value = false
    }
  }

  async function create(data: Record<string, unknown>): Promise<{ guest: Guest; is_duplicate: boolean }> {
    loading.value = true
    try {
      const res = await apiClient.post('/api/v1/guests', data)
      const guest = mapGuest(res.data.data)
      if (!res.data.is_duplicate) guests.value.unshift(guest)
      return { guest, is_duplicate: res.data.is_duplicate }
    } finally {
      loading.value = false
    }
  }

  async function setStatus(id: number, action: 'vip' | 'blacklist', payload: Record<string, unknown> = {}): Promise<void> {
    loading.value = true
    try {
      const res = await apiClient.post(`/api/v1/guests/${id}/${action}`, payload)
      const updated = mapGuest(res.data.data)
      const idx = guests.value.findIndex(g => g.id === id)
      if (idx !== -1) guests.value[idx] = updated
      if (current.value?.id === id) current.value = updated
    } finally {
      loading.value = false
    }
  }

  return { guests, current, loading, loadingList, loadingDetail, fetchAll, fetchOne, create, setStatus }
})
```
