### API Controller
- Namespace: App\Modules\[MODULE_NAME]\Controllers\Api\V1\[MODEL_NAME]Controller
- Thin — only: authorize, validate, call service, return response
- Constructor: private readonly [MODEL_NAME]Service $service
- index() calls: $this->service->paginate(filters, page, perPage)
- Response format: { status: 1|0, data: ..., message: string }
- Pagination response:
```json
{
  "status": 1,
  "data": {
    "items": [],
    "pagination": { "current_page": 1, "per_page": 15, "total": 0, "last_page": 1 }
  },
  "message": "..."
}
```
- HTTP codes: 200 / 201 / 204 / 422 / 404
- 404: catch ModelNotFoundException -> return status: 0

index() pattern:
```php
public function index(Request $request): JsonResponse
{
    $paginator = $this->service->paginate(
        filters:  $request->only(['search']),
        page:     (int) $request->get('page', 1),
        perPage:  (int) $request->get('per_page', 15)
    );

    return response()->json([
        'status' => 1,
        'data'   => [
            'items'      => [MODEL_NAME]Resource::collection($paginator),
            'pagination' => [
                'current_page' => $paginator->currentPage(),
                'per_page'     => $paginator->perPage(),
                'total'        => $paginator->total(),
                'last_page'    => $paginator->lastPage(),
            ],
        ],
        'message' => '[MODEL_NAME]s fetched successfully',
    ]);
}
```

### Pages — MANDATORY PATTERN
Before writing ANY .vue page:
1. MUST read `resources/js/Pages/Hotels/Create.vue` (highest priority — same module)
2. MUST read `resources/js/Pages/Reservations/Create.vue` (i18n pattern reference)
3. STRICTLY follow:

   **Permission (MANDATORY):**
   - `usePermission()` → `permission.check('create [MODEL_NAME_PLURAL]')`
   - `canCreate` computed → wrap entire form in `v-if="canCreate"`
   - `onMounted` → redirect to `/[ROUTE_SLUG]` if no permission

   **i18n Translations (MANDATORY):**
   - `import { useI18n } from '@/Composables/useI18n'`
   - `const { t } = useI18n()`
   - ALL user-facing strings MUST use `t('key')` — NO hardcoded strings
   - Labels, placeholders, button text, page title, hints — all via `t()`
   - `submitLabel` computed → `t('actions.creating')` / `t('actions.create')`

   **Form (MANDATORY):**
   - `useForm()` shape and field typing (with proper TS types e.g. `'' as number | string`)
   - `validateForm()` + `validateInertiaForm()` pattern
   - `form.clearErrors()` before submit
   - `form.setError()` for backend 422 errors
   - `mapBackendField()` for snake_case → camelCase error mapping
   - `scrollToFirstError()` after validation failure
   - `isSaving` computed from `form.processing || saving.value`
   - `Number(result.status) === 1` check
   - `form.reset()` before `router.visit()`
   - `watch()` for dependent field re-validation (if applicable)

4. DO NOT:
   - Use `reactive()` for form state
   - Use `reactive()` for errors
   - Skip `form.reset()` before `router.visit()`

- **Index.vue**  : search, status filter, `<Table>`, pagination, edit/delete per row
- **Create.vue** : `useForm()` + `usePermission()` + `useI18n()` — follow Hotels/Create.vue pattern

### Page Pattern (Create/Edit)
```ts
// ✅ Standard pattern — MUST match resources/js/Pages/Hotels/Create.vue
import { computed, watch, onMounted } from 'vue'
import { useForm, router } from '@inertiajs/vue3'
import { usePermission } from '@/Plugins/directives/permission'
import { useI18n } from '@/Composables/useI18n'
import { required, minValue, validateInertiaForm } from '@/Utils/validation'
import { use[MODEL_NAME]s } from '@/Composables/[MODULE_NAME]/use[MODEL_NAME]s'

// ─── i18n (MANDATORY — no hardcoded strings) ─────────────
const { t } = useI18n()

// ─── Permission (MANDATORY) ──────────────────────────────
const permission = usePermission()
const canCreate  = computed(() => permission.check('create [MODEL_NAME_PLURAL]'))

// ─── Composable ──────────────────────────────────────────
const { create, saving } = use[MODEL_NAME]s()

// ─── Form (MANDATORY — useForm, NEVER reactive) ──────────
const form = useForm({
    field1: '' as number | string,
    field2: '',
    // ... all [FIELD_LIST] fields with proper TS types
})

// ─── Computed ────────────────────────────────────────────
const isSaving    = computed(() => form.processing || saving.value)
const submitLabel = computed(() =>
    isSaving.value ? t('actions.creating') : t('actions.create')
)

// ─── watch for dependent field re-validation (if needed) ─
watch(() => form.dependentField, () => {
    if (form.otherField) validateOtherField()
})

// ─── onMounted permission guard ──────────────────────────
onMounted(() => {
    if (!canCreate.value) router.visit('/[ROUTE_SLUG]')
})

// ─── Submit ──────────────────────────────────────────────
async function submit(): Promise<void> {
    form.clearErrors()

    if (!validateForm()) {
        scrollToFirstError()
        return
    }

    try {
        const result = await create({
            field1: form.field1,
            field2: form.field2 || undefined,
        })

        if (Number(result.status) === 1) {
            form.reset()
            router.visit('/[ROUTE_SLUG]')
        }
    } catch (err: unknown) {
        const apiErr = err as Record<string, any>
        if (apiErr?.response?.data?.errors) {
            const backendErrors: Record<string, string[]> = apiErr.response.data.errors
            Object.entries(backendErrors).forEach(([key, messages]) => {
                form.setError(mapBackendField(key) as any, messages[0])
            })
            scrollToFirstError()
        }
    }
}
```

// ─── Validation ──────────────────────────────────────────
function validateForm(): boolean {
    return validateInertiaForm(form, {
        field1: [required],
        field2: [required, minValue(0)],
        // ... all required [FIELD_LIST] fields
    })
}

// ─── snake_case (Laravel) → camelCase (form) mapping ─────
function mapBackendField(field: string): string {
    const map: Record<string, string> = {
        field_one: 'fieldOne',
        field_two: 'fieldTwo',
        // ... all [FIELD_LIST] mappings
    }
    return map[field] ?? field
}

// ─── Scroll to first error ────────────────────────────────
function scrollToFirstError(): void {
    setTimeout(() => {
        const firstError = document.querySelector('.border-red-500')
        firstError?.scrollIntoView({ behavior: 'smooth', block: 'center' })
    }, 100)
}
```
